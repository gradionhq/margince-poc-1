# T11 Live UAT Guide

Run this only after the stack is up with `infra-up -> migrate-up -> seed-reset -> run`.
The commands below assume:

- the API is listening on `http://localhost:8080`
- the database is reachable through `DATABASE_URL`
- you can act as a tenant by sending `X-Workspace-Id`

Use one shell session so the variables set in the bootstrap block stay available for the later steps.

## Bootstrap

```bash
export API_BASE='http://localhost:8080'
export WS_ID='00000000-0000-0000-0000-000000000011'
export USER_ID='55555555-5555-5555-5555-555555555551'
export ROLE_ID='66666666-6666-6666-6666-666666666661'
export PIPELINE_A='11111111-1111-1111-1111-111111111111'
export PIPELINE_B='11111111-1111-1111-1111-111111111112'
export STAGE_A='22222222-2222-2222-2222-222222222221'
export STAGE_B='22222222-2222-2222-2222-222222222222'
export STAGE_OTHER='22222222-2222-2222-2222-222222222223'
export DEAL_NAME='T11 live UAT deal'
export REPLAY_KEY='t11-live-uat-replay'

psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO workspace (id, name, slug, base_currency) VALUES ('${WS_ID}', 'T11 live UAT workspace', 't11-live-uat', 'USD') ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO organization (id, workspace_id, name, source, captured_by) VALUES ('44444444-4444-4444-4444-444444444441', '${WS_ID}', 'T11 Partner Org', 'uat', 'human:t11') ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO person (id, workspace_id, full_name, source, captured_by) VALUES ('33333333-3333-3333-3333-333333333331', '${WS_ID}', 'T11 Stakeholder Person', 'uat', 'human:t11') ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO pipeline (id, workspace_id, name) VALUES ('${PIPELINE_A}', '${WS_ID}', 'T11 Pipeline A') ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO pipeline (id, workspace_id, name) VALUES ('${PIPELINE_B}', '${WS_ID}', 'T11 Pipeline B') ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO stage (id, workspace_id, pipeline_id, name, position) VALUES ('${STAGE_A}', '${WS_ID}', '${PIPELINE_A}', 'Qualify', 1) ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO stage (id, workspace_id, pipeline_id, name, position) VALUES ('${STAGE_B}', '${WS_ID}', '${PIPELINE_A}', 'Proposal', 2) ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO stage (id, workspace_id, pipeline_id, name, position) VALUES ('${STAGE_OTHER}', '${WS_ID}', '${PIPELINE_B}', 'Other Pipeline Stage', 1) ON CONFLICT (id) DO NOTHING;"

# RBAC bootstrap: the API authenticates via X-Workspace-ID + X-User-ID (legacy
# dev/test header path in workspaceWrap), and RbacMiddleware then loads
# permissions by joining role/role_assignment for that (workspace_id, user_id).
# Without a real app_user + role + role_assignment row here, every write below
# would get zero permissions.
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO app_user (id, workspace_id, email, display_name) VALUES ('${USER_ID}', '${WS_ID}', 't11-uat@example.com', 'T11 UAT User') ON CONFLICT (id) DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO role (id, workspace_id, key, is_system, permissions) VALUES ('${ROLE_ID}', '${WS_ID}', 't11_admin', true, '{\"deal\":{\"read\":{\"row_scope\":\"all\"},\"create\":{\"row_scope\":\"all\"},\"update\":{\"row_scope\":\"all\"},\"archive\":{\"row_scope\":\"all\"}},\"person\":{\"read\":{\"row_scope\":\"all\"}},\"organization\":{\"read\":{\"row_scope\":\"all\"}}}') ON CONFLICT (workspace_id, key) DO UPDATE SET permissions = EXCLUDED.permissions;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO role_assignment (workspace_id, role_id, user_id) VALUES ('${WS_ID}', '${ROLE_ID}', '${USER_ID}') ON CONFLICT (role_id, user_id, COALESCE(team_id, '00000000-0000-0000-0000-000000000000'::uuid)) DO NOTHING;"
```

## Step 1: Create a deal in a valid pipeline/stage pair

```bash
curl -i -sS \
  -H "Content-Type: application/json" \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  -H "Idempotency-Key: ${REPLAY_KEY}" \
  -d "{\"name\":\"${DEAL_NAME}\",\"pipeline_id\":\"${PIPELINE_A}\",\"stage_id\":\"${STAGE_A}\",\"source\":\"uat\",\"captured_by\":\"human:t11\"}" \
  "${API_BASE}/deals"

psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "SELECT from_stage_id, to_stage_id FROM deal_stage_history WHERE deal_id = (SELECT id FROM deal WHERE workspace_id = '${WS_ID}' AND name = '${DEAL_NAME}' ORDER BY created_at DESC LIMIT 1);"
```

Expected: `POST /deals` returns `201 Created` with a `Location` header, and `deal_stage_history` contains one row whose `from_stage_id` is `NULL` and whose `to_stage_id` matches `${STAGE_A}`.

## Step 2: Replay the same create with the same idempotency key

```bash
export DEAL_ID="$(psql "$DATABASE_URL" -At -v ON_ERROR_STOP=1 -c "SELECT id FROM deal WHERE workspace_id = '${WS_ID}' AND name = '${DEAL_NAME}' ORDER BY created_at DESC LIMIT 1;")"

curl -i -sS \
  -H "Content-Type: application/json" \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  -H "Idempotency-Key: ${REPLAY_KEY}" \
  -d "{\"name\":\"${DEAL_NAME}\",\"pipeline_id\":\"${PIPELINE_A}\",\"stage_id\":\"${STAGE_A}\",\"source\":\"uat\",\"captured_by\":\"human:t11\"}" \
  "${API_BASE}/deals"

psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "SELECT count(*) FROM deal WHERE workspace_id = '${WS_ID}' AND name = '${DEAL_NAME}';"
```

Expected: the replay returns the same created deal payload and `SELECT count(*)` still reports `1`.

## Step 3: Create with a stage from a different pipeline

```bash
curl -i -sS \
  -H "Content-Type: application/json" \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  -d "{\"name\":\"Bad stage deal\",\"pipeline_id\":\"${PIPELINE_A}\",\"stage_id\":\"${STAGE_OTHER}\",\"source\":\"uat\",\"captured_by\":\"human:t11\"}" \
  "${API_BASE}/deals"
```

Expected: `422 Unprocessable Entity` with `details.errors` containing exactly `{field:"stage_id", code:"stage_not_in_pipeline"}`.

## Step 4: Update with a stale `If-Match`

```bash
curl -i -sS -X PATCH \
  -H "Content-Type: application/json" \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  -H "If-Match: 999" \
  -d "{\"name\":\"${DEAL_NAME} stale update\"}" \
  "${API_BASE}/deals/${DEAL_ID}"
```

Expected: `409 Conflict` with problem code `version_skew`.

## Step 5: Update with a malformed `If-Match`

```bash
curl -i -sS -X PATCH \
  -H "Content-Type: application/json" \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  -H "If-Match: not-a-number" \
  -d "{\"name\":\"${DEAL_NAME} malformed update\"}" \
  "${API_BASE}/deals/${DEAL_ID}"
```

Expected: `400 Bad Request`.

## Step 6: List deals with the DM-VOCAB-3 filter/sort fields

```bash
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "INSERT INTO relationship (id, workspace_id, deal_id, person_id, kind, role, source, captured_by, archived_at) VALUES (uuidv7(), '${WS_ID}', '${DEAL_ID}', '33333333-3333-3333-3333-333333333331', 'deal_stakeholder', 'champion', 'uat', 'human:t11', NULL) ON CONFLICT DO NOTHING;"
psql "$DATABASE_URL" -v ON_ERROR_STOP=1 -c "UPDATE deal SET status = 'open', forecast_category = 'commit', partner_org_id = '44444444-4444-4444-4444-444444444441', amount_minor = 2500, currency = 'USD' WHERE id = '${DEAL_ID}' AND workspace_id = '${WS_ID}';"
curl -i -sS \
  -H "Content-Type: application/json" \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  -H "Idempotency-Key: t11-live-uat-list-second" \
  -d "{\"name\":\"Later deal\",\"pipeline_id\":\"${PIPELINE_A}\",\"stage_id\":\"${STAGE_A}\",\"amount_minor\":9000,\"currency\":\"USD\",\"status\":\"open\",\"forecast_category\":\"commit\",\"partner_org_id\":\"44444444-4444-4444-4444-444444444441\",\"source\":\"uat\",\"captured_by\":\"human:t11\"}" \
  "${API_BASE}/deals"

curl -sS \
  -H "X-Workspace-Id: ${WS_ID}" \
  -H "X-User-ID: ${USER_ID}" \
  "${API_BASE}/deals?sort=amount_minor&status=open&forecast_category=commit&partner_org_id=44444444-4444-4444-4444-444444444441" \
  | sed -n '1,200p'
```

Expected: `GET /deals` returns `200 OK`, only the matching rows are present, the rows are sorted by `amount_minor`, and the payload includes `stage_entered_at` and `stakeholder_count`.
